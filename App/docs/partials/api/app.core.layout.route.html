service in module app.core.layout

Description

app.layout.route

The app.layout.route module is a Cross App Config Object which allows you the specify the default layout with a nav-view.

Other module templates will be loaded into the nav-view in the layout.html file.

Dependencies

  • app
  • layout
  • $stateProvider

Example

(function() {
  'use strict';

  angular
    .module('app.layout')
    .config(route);
  route.$inject = ['$stateProvider'];

  function route($stateProvider) {
    $stateProvider
      .state('app', {
        url: '/app',
        abstract: true,
        templateUrl: '_core/layout/layout.html',
        controller: 'LayoutCtrl',
        controllerAs: 'layout'
      });
  }
})();
