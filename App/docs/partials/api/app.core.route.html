route
service in module app.core

Description

app.core.route

The app.core.route module is a Cross App Config Object which allows you the specify the default route if a valid route is not defined by the user.

Dependencies

  • app
  • core
  • constants

Example

(function() {
  'use strict';
  angular
  .module('app.core')
  .config(route);
  // Inject the $urlRouterProvider
  route.$inject = ['$urlRouterProvider', CORECONSTANTS];

  // Set the $urlRouterProvider.otherwise to the default state defined in the CORECONSTANTS.DEFAULT_URL
  function route($urlRouterProvider, CORECONSTANTS) {
    $urlRouterProvider.otherwise( CORECONSTANTS.DEFAULT_URL );
  }
})();
