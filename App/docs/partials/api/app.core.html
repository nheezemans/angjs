app.core

This module allows to include other Angular Modules, Cross App Modules & 3rd Party Modules to be included in the App

Example

(function() {
  'use strict';

  angular
  .module('app.core', [
    // Angular Modules

    // Cross App Modules
    // Example to load a module called common.device:
    'app.core.models',
    'app.core.services',

    //3rd-party Modules

  ]);

})(); // End Strict
