route
service in module app.dashboard

Description

app.dashboard.route

The app.dashboard.route module is the route, views, controllers and templates used by this module.

Dependencies

  • app
  • $stateProvider

Example

(function() {
  'use strict';

  angular
    .module('app.dashboard')
    .config(route);
  route.$inject = ['$stateProvider'];

  function route($stateProvider) {
    $stateProvider
    .state('app.dashboard', {
      url: '/dashboard',
      views: {
        'appContent': {
          templateUrl: 'dashboard/dashboard.html',
          controller: 'DashboardCtrl',
          controllerAs: 'dashboard'
        }
      }
    });
  }
})();
