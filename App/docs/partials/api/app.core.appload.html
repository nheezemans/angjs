appload
service in module app.core

Description

app.core.appload.route

The app.core.appload.route module is the route, views, controllers and templates used by this module.

Dependencies

  • app
  • $stateProvider

Example

(function() {
  'use strict';

  angular
    .module('app.core.appload')
    .config(route);
  route.$inject = ['$stateProvider'];

  function route($stateProvider) {
    $stateProvider
    .state('appload', {
      url: '/appload',
      templateUrl: '_core/appload/appload.html',
      controller: 'ApploadCtrl',
      controllerAs: 'appload'
    });
  }
})();
